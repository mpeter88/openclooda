## Problem

The proposals system (`proposals.ts`, `PolicyProposal` type, `countPending()`) exists
and the `before_agent_start` notice already fires when proposals are pending. But only
the **Meta-Reviewer** currently writes proposals, and only for SOUL.md policy changes.

The Archivist runs every 10 turns with access to all episodic patterns. It's the right
place to generate _actionable_ proposals — not just policy suggestions, but:

- "This pattern keeps recurring — here's what I'd do about it"
- "You've done this manually three times — here's a script that automates it"
- "This architectural gap keeps causing failures — here's the fix"

This is the first genuinely proactive capability: initiative based on accumulated
context rather than reactive response to prompts.

### P2 — Proposal extraction in Archivist

**File:** `extensions/memory-ooda/archivist.ts`

Add a second model pass after pattern extraction. Runs only when there are
enough events to identify a pattern (≥ 5 events):

```typescript
/** Represents a candidate proposal from the Archivist. */
interface ProposalCandidate {
  category: "project" | "workflow" | "technical";
  rule: string; // domain label (e.g. "AMF Pipeline", "OpenCLOODA")
  proposal: string; // concrete actionable suggestion (1-2 sentences)
  reasoning: string; // why this matters
  evidence: string[]; // 2-3 specific examples from events
  confidence: number; // 0.0–1.0
}

async function extractProposals(
  events: EpisodicEvent[],
  existingKnowledge: KnowledgeFile,
  existingProposals: PolicyProposal[],
  callModel: ModelCallFn,
): Promise<ProposalCandidate[]> {
  if (events.length < 5) return [];

  const eventSummary = events
    .slice(0, 50) // cap context
    .map((e) => `[${e.category}] ${e.text.slice(0, 200)}`)
    .join("\n");

  const existingPending = existingProposals
    .filter((p) => p.status === "pending")
    .map((p) => p.proposal)
    .join("\n");

  const prompt = `
You are scanning recent session memory for actionable suggestions worth surfacing.

## Recent Events (${events.length} total, showing first 50):
${eventSummary}

## Already Pending (do NOT duplicate):
${existingPending || "(none)"}

## What to look for:

RECURRING PROBLEMS — same issue appeared 3+ times:
  → Suggest a fix, process change, or automation

MISSING TOOLS — task done manually that could be scripted:
  → Suggest building it (script, CLI command, automation)

ARCHITECTURAL GAPS — pattern of workarounds for a known root cause:
  → Suggest addressing the root cause directly

PROCESS INEFFICIENCY — repeated friction in the workflow:
  → Suggest streamlining

## Rules:
- Only emit proposals with confidence >= 0.6
- Max 3 proposals per run
- Each must be concrete and actionable (not vague)
- Do NOT propose things already addressed in pending proposals
- Do NOT propose things clearly outside the agent's scope

## Output:
JSON array of proposals (empty array if nothing warrants a proposal):
[
  {
    "category": "project" | "workflow" | "technical",
    "rule": "<domain label>",
    "proposal": "<1-2 sentences: concrete action>",
    "reasoning": "<why this matters>",
    "evidence": ["<example 1>", "<example 2>"],
    "confidence": 0.0-1.0
  }
]
`;

  try {
    const raw = await callModel(prompt);
    const parsed = JSON.parse(stripCodeFences(raw));
    if (!Array.isArray(parsed)) return [];

    return parsed.filter(
      (p: ProposalCandidate) =>
        p.confidence >= 0.6 && typeof p.proposal === "string" && p.proposal.trim().length > 0,
    );
  } catch {
    return [];
  }
}
```

### P3 — Semantic dedup for proposals

**File:** `extensions/memory-ooda/proposals.ts`

Before writing a new proposal, check if a semantically similar one already exists
among pending proposals. Use simple word-overlap ratio (no embeddings needed here
— proposals are short and distinct):

```typescript
function overlapRatio(a: string, b: string): number {
  const wordsA = new Set(a.toLowerCase().split(/\W+/).filter(Boolean));
  const wordsB = new Set(b.toLowerCase().split(/\W+/).filter(Boolean));
  const intersection = [...wordsA].filter((w) => wordsB.has(w)).length;
  return intersection / Math.max(wordsA.size, wordsB.size);
}

export function isDuplicateProposal(
  candidate: string,
  existing: PolicyProposal[],
  threshold = 0.5,
): boolean {
  return existing
    .filter((p) => p.status === "pending")
    .some((p) => overlapRatio(candidate, p.proposal) >= threshold);
}

export function addArchivistProposals(
  workspacePath: string,
  candidates: ProposalCandidate[],
): number {
  const existing = getProposals(workspacePath);
  const pendingCount = existing.filter((p) => p.status === "pending").length;

  // Hard cap: max 5 total pending proposals at any time
  if (pendingCount >= 5) return 0;

  let added = 0;
  for (const c of candidates) {
    if (pendingCount + added >= 5) break;
    if (isDuplicateProposal(c.proposal, existing)) continue;

    existing.push({
      id: crypto.randomUUID(),
      timestamp: new Date().toISOString(),
      rule: c.rule,
      proposal: c.proposal,
      reasoning: c.reasoning,
      evidence: c.evidence,
      status: "pending",
      category: c.category,
      confidence: c.confidence,
      autoGenerated: true,
    });
    added++;
  }

  if (added > 0) {
    writeProposals(workspacePath, existing);
  }

  return added;
}
```

if (events.length >= 5) {
  try {
    const existingProposals = getProposals(workspacePath);
    const candidates = await extractProposals(events, knowledge, existingProposals, callModel);
    const added = addArchivistProposals(workspacePath, candidates);
    if (added > 0) {
      store.appendArchivistLog(
        "proposals_generated",
        `Generated ${added} proposal(s) from ${events.length} events`,
      );
    }
  } catch (err) {
    // Non-fatal — proposals are best-effort

  }
}
